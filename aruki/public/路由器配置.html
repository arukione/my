配置php路径

vim /etc/config/uhttpd

config uhttpd web
    list    listen_http 0.0.0.0:88
    option  home    /www
    option  cgi_prefix  /cgi-bin
    option  index_page  "index.html index.php"
    list    interpreter ".php=/usr/bin/php-cgi"

重新加载 /etc/init.d/uhttpd restart
